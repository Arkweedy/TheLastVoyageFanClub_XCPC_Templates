\subsubsection*{$\cdot$ Integrals involving $ax^2+bx+c$ ($a>0$)}
\[ 
\int \frac{\md x}{ax^2+bx+c} = \begin{cases}
\frac{2}{\sqrt{4ac-b^2}}\arctan\frac{2ax+b}{\sqrt{4ac-b^2}} + C & (b^2 < 4ac) \\[2ex]
\frac{1}{\sqrt{b^2-4ac}}\ln\left| \frac{2ax+b-\sqrt{b^2-4ac}}{2ax+b+\sqrt{b^2-4ac}} \right| + C & (b^2 > 4ac)
\end{cases} 
\]
\[ 
\int \frac{x}{ax^2+bx+c} \md x = \frac{1}{2a} \ln |ax^2+bx+c| - \frac{b}{2a} \int \frac{\md x}{ax^2+bx+c} 
\]

\subsubsection*{$\cdot$ Integrals involving $\sqrt{\pm ax^2+bx+c}$ ($a>0$)}
\[ 
\int \frac{\md x}{\sqrt{ax^2+bx+c}} = \frac{1}{\sqrt{a}} \ln \left| 2ax+b+2\sqrt{a}\sqrt{ax^2+bx+c} \right| + C 
\]
\[ 
\begin{split}
\int \sqrt{ax^2+bx+c} \md x = {}& \frac{2ax+b}{4a}\sqrt{ax^2+bx+c} \\
& + \frac{4ac-b^2}{8a\sqrt{a}}\ln \left|2ax+b+2\sqrt{a}\sqrt{ax^2+bx+c}\right| + C 
\end{split}
\]
\[ 
\begin{split}
\int \frac{x}{\sqrt{ax^2+bx+c}} \md x = {}& \frac{1}{a}\sqrt{ax^2+bx+c} \\
& - \frac{b}{2a\sqrt{a}}\ln \left| 2ax+b+2\sqrt{a}\sqrt{ax^2+bx+c} \right| + C 
\end{split}
\]
\[ 
\int \frac{\md x}{\sqrt{c+bx-ax^2}} = \frac{1}{\sqrt{a}} \arcsin \frac{2ax-b}{\sqrt{b^2+4ac}} + C  
\]
\[ 
\begin{split}
\int \sqrt{c+bx-ax^2} \md x = {}& \frac{2ax-b}{4a}\sqrt{c+bx-ax^2} \\
& + \frac{b^2+4ac}{8a\sqrt{a}}\arcsin\frac{2ax-b}{\sqrt{b^2+4ac}} + C 
\end{split}
\]
\[ 
\int \frac{x}{\sqrt{c+bx-ax^2}} \md x = -\frac{1}{a}\sqrt{c+bx-ax^2} + \frac{b}{2a\sqrt{a}}\arcsin\frac{2ax-b}{\sqrt{b^2+4ac}} + C 
\]


\subsubsection*{$\cdot$ Integrals involving $\sqrt{(x-a)(b-x)}$, ($a<b$)}
\[ 
\int \frac{\md x}{\sqrt{(x-a)(b-x)}} = 2\arcsin\sqrt{\frac{x-a}{b-a}} + C
\]
\[ 
\begin{split}
\int \sqrt{(x-a)(b-x)} \md x = {}& \frac{2x-a-b}{4}\sqrt{(x-a)(b-x)} \\
& + \frac{(b-a)^2}{8}\arcsin\frac{2x-a-b}{b-a} + C
\end{split}
\]

\subsubsection*{Trigonometric Integrals (Constant $+C$ is omitted)}
\begin{multicols}{2}
    $\int \tan x \md x = -\ln|\cos x|$
    \\ $\int \cot x \md x = \ln |\sin x|$
    \\ $\int \sec x \md x = \ln |\sec x + \tan x|$
    \\ $\int \csc x \md x = \ln |\csc x - \cot x|$
    \\ $\int \sec^2 x \md x = \tan x$
    \\ $\int \csc^2 x \md x = -\cot x$
    \\ $\int \sec x \tan x \md x = \sec x$
    \\ $\int \csc x \cot x \md x = -\csc x$
    \\ $\int \sin^2 x \md x = \frac{x}{2} - \frac{1}{4} \sin 2x$
    \\ $\int \cos^2 x \md x = \frac{x}{2} + \frac{1}{4} \sin 2x$
\end{multicols}
% Reduction formulas
\[ 
\int \sin^n x \md x = -\frac{1}{n} \sin^{n-1} x \cos x + \frac{n-1}{n} \int \sin^{n-2} x \md x 
\]
\[ 
\int \cos^n x \md x = \frac{1}{n} \cos^{n-1} x \sin x + \frac{n-1}{n} \int \cos^{n-2} x \md x 
\]
\[ 
\int \frac{\md x}{\sin^n x} = -\frac{\cos x}{(n-1)\sin^{n-1}x} + \frac{n-2}{n-1} \int \frac{\md x}{\sin^{n-2}x} 
\]
\[ 
\int \frac{\md x}{\cos^n x} = \frac{\sin x}{(n-1)\cos^{n-1}x} + \frac{n-2}{n-1} \int \frac{\md x}{\cos^{n-2}x} 
\]
\[ 
\begin{split} \int \cos^m x \sin^n x \md x 
    = {}& \frac{\cos^{m-1} x \sin^{n+1}x}{m+n} + \frac{m-1}{m+n}\int\cos^{m-2}x\sin^nx\md x \\
    = {}& -\frac{\cos^{m+1} x \sin^{n-1}x}{m+n} + \frac{n-1}{m+n} \int \cos^m x\sin^{n-2} x \md x 
\end{split} 
\]
\[ 
\int \frac{\md x}{a + b \sin x} = \begin{cases}
    \frac{2}{\sqrt{a^2-b^2}}\arctan\frac{a\tan\frac{x}{2}+b}{\sqrt{a^2-b^2}} & (a^2 > b^2) \\[2ex]
    \frac{1}{\sqrt{b^2-a^2}}\ln \left| \frac{a\tan\frac{x}{2}+b-\sqrt{b^2-a^2}}{a\tan\frac{x}{2}+b+\sqrt{b^2-a^2}} \right| & (a^2 < b^2)
    \end{cases} 
\]
\[ 
\int \frac{\md x}{a + b \cos x} = \begin{cases}
    \frac{2}{\sqrt{a^2-b^2}}\arctan\left(\sqrt{\frac{a-b}{a+b}}\tan\frac{x}{2}\right) & (a^2 > b^2) \\[2ex]
    \frac{1}{\sqrt{b^2-a^2}} \ln \left| \frac{\tan\frac{x}{2}\sqrt{b+a}+\sqrt{b-a}}{\tan\frac{x}{2}\sqrt{b+a}-\sqrt{b-a}} \right| & (a^2 < b^2)
    \end{cases} 
\]
\[ 
\int \frac{\md x}{a^2\cos^2x+b^2\sin^2x} = \frac{1}{ab} \arctan\left( \frac{b}{a}\tan x \right) 
\]
\[ 
\int \frac{\md x}{a^2\cos^2x-b^2\sin^2x} = \frac{1}{2ab}\ln\left|\frac{b\tan x+a}{b\tan x-a}\right| 
\]
\[ 
\int x \sin (ax) \md x = \frac{1}{a^2} \sin (ax) - \frac{1}{a} x \cos (ax) 
\]
\[ 
\int x^2 \sin (ax) \md x = -\frac{1}{a} x^2 \cos (ax) + \frac{2}{a^2} x \sin (ax) + \frac{2}{a^3} \cos (ax) 
\]
\[ 
\int x \cos (ax) \md x = \frac{1}{a^2} \cos (ax) + \frac{1}{a} x \sin (ax) 
\]
\[ 
\int x^2 \cos (ax) \md x = \frac{1}{a} x^2 \sin (ax) + \frac{2}{a^2} x \cos (ax) - \frac{2}{a^3} \sin (ax) 
\]

\subsubsection*{Inverse Trigonometric Integrals ($a>0$)}
\[ 
\int \arcsin \frac{x}{a} \md x = x \arcsin \frac{x}{a} + \sqrt{a^2-x^2}+C 
\]
\[ 
\int x \arcsin \frac{x}{a} \md x= \left(\frac{x^2}{2}-\frac{a^2}{4}\right)\arcsin \frac{x}{a} + \frac{x}{4} \sqrt{a^2-x^2}+C
\]
\[ 
\int x^2 \arcsin \frac{x}{a} \md x = \frac{x^3}{3}\arcsin \frac{x}{a}+\frac{1}{9}(x^2+2 a^2)\sqrt{a^2-x^2}+C 
\]
\[ 
\int \arccos \frac{x}{a} \md x= x \arccos \frac{x}{a} - \sqrt{a^2-x^2} +C 
\]
\[ 
\int x \arccos \frac{x}{a} \md x= \left(\frac{x^2}{2}-\frac{a^2}{4}\right)\arccos \frac{x}{a} - \frac{x}{4} \sqrt{a^2-x^2}+C 
\]
\[ 
\int x^2 \arccos \frac{x}{a}\md x= \frac{x^3}{3}\arccos \frac{x}{a} - \frac{1}{9}(x^2+2a^2)\sqrt{a^2-x^2}+C
\]
\[ 
\int \arctan \frac{x}{a} \md x=x \arctan \frac{x}{a}-\frac{a}{2}\ln (a^2+x^2)+C 
\]
\[ 
\int x\arctan \frac{x}{a} \md x = \frac{1}{2}(x^2+a^2)\arctan \frac{x}{a} -\frac{ax}{2}+C 
\]
\[ 
\int x^2 \arctan \frac{x}{a} \md x= \frac{x^3}{3} \arctan \frac{x}{a} - \frac{ax^2}{6} + \frac{a^3}{6} \ln (a^2+x^2)+C 
\]

\subsubsection*{Exponential Integrals}
\[ 
\int a^x \md x= \frac{a^x}{\ln a} + C
\]
\[ 
\int \me^{ax}\md x=\frac{1}{a}\me^{ax}+C 
\] 
\[ 
\int x \me^{ax} \md x=\frac{\me^{ax}}{a^2}(ax-1) +C 
\]
\[ 
\int x^n \me^{ax} \md x=\frac{1}{a}x^n \me^{ax}-\frac{n}{a} \int x^{n-1} \me^{ax} \md x 
\]
\[ 
\int x a^x \md x = \frac{xa^x}{\ln a}-\frac{a^x}{(\ln a)^2}+C 
\]
\[ 
\int x^n a^x \md x= \frac{x^n a^x}{\ln a}-\frac{n}{\ln a}\int x^{n-1}a^x \md x 
\]
\[ 
\int \me^{ax} \sin(bx) \md x = \frac{\me^{ax}}{a^2+b^2}(a \sin(bx) - b \cos(bx))+C 
\]
\[ 
\int \me^{ax} \cos(bx) \md x = \frac{\me^{ax}}{a^2+b^2}(a \cos(bx) + b \sin(bx))+C 
\]
\[ 
\begin{split}
\int \me^{ax} \sin^n (bx) \md x = {}& \frac{\me^{ax} \sin^{n-1}(bx)}{a^2+b^2 n^2} (a \sin(bx) - nb \cos(bx)) \\
& +\frac{n(n-1)b^2}{a^2+b^2 n^2}\int \me^{ax} \sin^{n-2}(bx) \md x 
\end{split}
\]
\[ 
\begin{split}
\int \me^{ax} \cos^n (bx) \md x = {}& \frac{\me^{ax} \cos^{n-1}(bx)}{a^2+b^2 n^2} (a \cos(bx) + nb \sin(bx)) \\
& +\frac{n(n-1)b^2}{a^2+b^2 n^2}\int \me^{ax} \cos^{n-2}(bx) \md x 
\end{split}
\]

\subsubsection*{Logarithmic Integrals}
\[ 
\int \ln x \md x = x \ln x - x + C
\]
\[ 
\int \frac{\md x}{x \ln x} =\ln |\ln x|+C 
\]
\[ 
\int x^n \ln x \md x = \frac{x^{n+1}}{n+1}\left(\ln x - \frac{1}{n+1} \right) +C 
\]
\[ 
\int (\ln x)^{n} \md x = x(\ln x)^n - n \int (\ln x)^{n-1} \md x 
\]
\[ 
\int x^m(\ln x)^n \md x=\frac{x^{m+1}(\ln x)^n}{m+1} - \frac{n}{m+1} \int x^m(\ln x)^{n-1}\md x 
\]
